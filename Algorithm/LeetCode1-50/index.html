 * struct ListNode {

 * };

class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* l0;

        ListNode* head=new ListNode(-1);
        l0=head;
        while(l1!=NULL&&l2!=NULL)
        {
            if(l1->val<=l2->val) 
            {
                l0->next=new ListNode(l1->val);
                l1=l1->next;
            }
            else 
            {
                l0->next=new ListNode(l2->val);
                l2=l2->next;
            }
            l0=l0->next;
        }
        l0->next = l1 ? l1 : l2;
        return head->next;
    }
};
